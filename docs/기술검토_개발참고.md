```python
# backend/services/analysis_cache.py
import hashlib
from cachetools import TTLCache

_cache: TTLCache = TTLCache(maxsize=100, ttl=86400)

def get_cache_key(image_bytes: bytes) -> str:
  return hashlib.sha256(image_bytes).hexdigest()

async def get_or_analyze(image_bytes: bytes, provider: LLMProvider) -> AnalysisResult:
  key = get_cache_key(image_bytes)
  if key in _cache:
    return _cache[key]
  result = await provider.analyze_floorplan(image_bytes)
  _cache[key] = result
  return result
```

## 3. 추가 기술 검토 사항

### TECH-01 🟢 SVG XSS 취약점 (Phase 2, SVG 배치도 도입 시)

**문제**
MVP에서는 SVG 배치도를 생성하지 않는다. Phase 2에서 Claude가 생성한 SVG XML을 `dangerouslySetInnerHTML`로 렌더링하면 XSS 가능.
`<script>`, `<foreignObject>`, `onload="..."` 등 JS 실행 벡터가 SVG에 존재.

**프론트엔드 — DOMPurify Sanitization**

```bash
npm install dompurify @types/dompurify
```

```tsx
// frontend/components/result/SvgLayoutViewer.tsx
import DOMPurify from 'dompurify'

const ALLOWED_SVG_TAGS = ['svg', 'rect', 'text', 'line', 'path', 'circle', 'g', 'defs', 'symbol', 'use']

interface SvgLayoutViewerProps {
  svgXml: string
}

export function SvgLayoutViewer({ svgXml }: SvgLayoutViewerProps) {
  const sanitized = DOMPurify.sanitize(svgXml, {
    USE_PROFILES: { svg: true, svgFilters: true },
    ALLOWED_TAGS: ALLOWED_SVG_TAGS,
    FORBID_ATTR: ['onload', 'onclick', 'onerror', 'href', 'xlink:href'],
  })

  return (
    <div
      className='w-full aspect-square border rounded-lg overflow-hidden'
      dangerouslySetInnerHTML={{ __html: sanitized }}
    />
  )
}
```

**백엔드 — 2차 검증 (Claude 출력 직후)**

```python
# backend/services/svg_service.py
import re

FORBIDDEN_SVG_PATTERNS = [
  r'<script',
  r'javascript:',
  r'on\w+\s*=',
  r'<foreignObject',
  r'data:text/html',
  r'<iframe',
]

def validate_and_sanitize_svg(svg_xml: str) -> str:
  """Claude 생성 SVG 보안 검증. 위험 패턴 발견 시 ValueError."""
  for pattern in FORBIDDEN_SVG_PATTERNS:
    if re.search(pattern, svg_xml, re.IGNORECASE):
      raise ValueError(f'SVG 보안 검증 실패: {pattern}')

  # viewBox 없으면 추가 (렌더링 보정)
  if 'viewBox' not in svg_xml:
    svg_xml = svg_xml.replace('<svg', '<svg viewBox="0 0 400 400"', 1)

  return svg_xml
```

```python
import imghdr
from fastapi import UploadFile, HTTPException

ALLOWED_MIME = {'jpeg', 'png'}
MAX_FILE_BYTES = 5 * 1024 * 1024  # 5MB

async def validate_upload(file: UploadFile) -> bytes:
  raw = await file.read()

  if len(raw) > MAX_FILE_BYTES:
    raise HTTPException(413, f'파일 크기 제한 초과 (최대 5MB, 현재 {len(raw) // 1024}KB)')

  # imghdr는 Content-Type 헤더가 아닌 실제 파일 바이트를 검사
  detected = imghdr.what(None, raw)
  if detected not in ALLOWED_MIME:
    raise HTTPException(
      415,
      f'JPG 또는 PNG 파일만 허용됩니다. (감지된 타입: {detected or "알 수 없음"})'
    )

  return raw

async def analyze(
```

### TECH-03 🔴 20~25초 대기 UX — SSE 진행률 스트리밍

**문제**
`/render` 단일 응답으로 방별 이미지 생성을 기다리면 발표 중 "서비스 멈춤"으로 오해.
Vercel 함수 60초 제한 내에는 들어오지만 사용자 피드백이 전무.

**백엔드 — SSE 스트리밍**

```python
# backend/routers/render.py
from fastapi.responses import StreamingResponse
import json

@router.post('/render/stream')
async def render_stream(body: RenderRequest, user=Depends(verify_jwt)):

  async def event_generator():
    try:
      # 단계 1: 선택 톤과 방 목록 검증
      yield f"data: {json.dumps({'step': 'preparing', 'pct': 10, 'message': '방별 시안 준비 중...'})}\n\n"

      rooms = await load_render_targets(body.session_id)
      yield f"data: {json.dumps({'step': 'rendering', 'pct': 35, 'message': '방별 이미지 생성 중...'})}\n\n"

      # 단계 2: Imagen + Naver 병렬
      renders, products = await asyncio.gather(
        imagen_service.generate_by_room(body.selected_tone_id, rooms),
        naver_service.search_by_room(body.selected_tone_id, rooms),
        return_exceptions=True
      )
      yield f"data: {json.dumps({'step': 'saving', 'pct': 90, 'message': '결과 저장 중...'})}\n\n"

      result = await save_result(body, rooms, renders, products, user.id)
      yield f"data: {json.dumps({'step': 'done', 'pct': 100, 'result_id': str(result.id)})}\n\n"

    except Exception as e:
      yield f"data: {json.dumps({'step': 'error', 'message': str(e)})}\n\n"

  return StreamingResponse(event_generator(), media_type='text/event-stream')
```

**프론트엔드 — 진행률 UI**

```tsx
// frontend/app/result/[id]/page.tsx
'use client'
import { useState, useEffect } from 'react'
import { Progress } from '@/components/ui/progress'

const STEP_LABELS: Record<string, string> = {
  preparing:  '방별 시안 준비 중...',
  rendering:  '방별 이미지 생성 중...',
  saving:     '결과 저장 중...',
  done:       '완료!',
}

export default function ResultPage() {
  const [progress, setProgress] = useState(0)
  const [stepLabel, setStepLabel] = useState('')
  const [resultId, setResultId] = useState<string | null>(null)

  useEffect(() => {
    const source = new EventSource(`${process.env.NEXT_PUBLIC_API_URL}/render/stream`, {
      // POST body는 별도 fetch + ReadableStream으로 처리
    })

    source.onmessage = (e) => {
      const data = JSON.parse(e.data)
      setProgress(data.pct)
      setStepLabel(STEP_LABELS[data.step] ?? '')

      if (data.step === 'done') {
        setResultId(data.result_id)
        source.close()
      }
      if (data.step === 'error') {
        source.close()
      }
    }

    return () => source.close()
  }, [])

  if (!resultId) {
    return (
      <div className='flex flex-col items-center gap-4 p-8'>
        <p className='text-sm text-muted-foreground'>{stepLabel}</p>
        <Progress value={progress} className='w-full max-w-md' />
        <p className='text-xs text-muted-foreground'>{progress}%</p>
      </div>
    )
  }

  return <ResultContent resultId={resultId} />
}
```

> SSE는 단방향이라 POST body를 직접 전달할 수 없습니다.
> 실제 구현 시 `fetch` + `ReadableStream`으로 처리하거나,
> POST `/render` → `job_id` 반환 → GET `/render/{job_id}/stream` SSE 폴링 방식도 가능합니다.

### TECH-08 🟡 Supabase RLS 미설정

**문제**
FastAPI JWT 검증만 있고 DB 레벨 접근 제어 없음.
백엔드 코드 버그 발생 시 다른 사용자 데이터 조회 가능.

**보완책 — Supabase SQL Editor에서 실행**

```sql
-- 1. analysis_sessions: 본인 세션만 접근
alter table analysis_sessions enable row level security;
create policy "본인 세션만 접근" on analysis_sessions
  using (user_id = auth.uid());

-- 2. recommendation_results: 본인 결과만 접근
alter table recommendation_results enable row level security;
create policy "본인 결과만 접근" on recommendation_results
  using (user_id = auth.uid());

-- 3. share_links: 누구나 읽기 가능 (공유 목적)
alter table share_links enable row level security;
create policy "공유 링크 공개 읽기" on share_links
  for select using (true);

-- 4. room_renders / products: recommendation_results 통해 간접 접근
--    별도 RLS 없이 FK CASCADE로 보호됨
```

> FastAPI는 `SUPABASE_SERVICE_ROLE_KEY`로 RLS를 우회하므로 서버 로직에 영향 없음.
> 이 설정은 대시보드 직접 접근이나 anon key 노출 시 방어선.
